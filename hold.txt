class Circle{
    constructor(xpos, ypos, radius, color, speed){
        this.xpos = xpos
        this.ypos = ypos
        this.radius = radius
        this.color = color
        this.speed = speed
        this.dx = 1 + this.speed
        this.dy = 1 + this.speed
    }
    draw(context) {
        context.beginPath()
        context.lineWidth = 5
        context.arc(this.xpos, this.ypos, this.radius, 0 , Math.PI*2, false)
        context.fillStyle = this.color
        context.fill()
        context.stroke()
        context.closePath()   
    }
    bounce(){
        this.draw(context)

        if ((this.ypos + this.radius) > canvas.height){//bottom
            this.dy = -this.dy
           }
           if((this.ypos - this.radius)< 0){ //top
            this.dy = -this.dy
           }

        this.xpos += this.dx
        this.ypos += this.dy
    }

}
